#==================================================
# resume-job-matcher - FastAPI backend Dockerfile
# Multi-stage build > final image < 200 MB
#==================================================

# Stage 1: Install dependencies
FROM python:3.12-slim AS builder
WORKDIR /app

# Copy only requirements first (caching magic)
COPY ../backend/requirements.txt
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# Stage 2: Final tiny image
FROM python:3.12-slim

# Create non-root user (security point)
RUN adduser --disabled-password --gecos '' appuser
WORKDIR /app

# Copy installed packages from builder
COPY --from=builder /usr/local/lib/python3.12/sit-packages /user/local/lib/python3.12/site-packages

# Copy only the code we need
COPY ../backend/ .

# Switch to non-root user
USER appuser

# Export port 
EXPOSE 8000

# Run FastAPI with Uvicorn
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
